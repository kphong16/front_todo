</head>
<body>
    <div class="container">
        <h1>TODO 목록 앱</h1>
        <p>
            <!-- Add Todo link (Update with actual link if needed) -->
            <a href=""><i class="bi-plus"></i>Add Todo</a>
            <!-- Link to completed todos (Update with actual link if needed) -->
            <a href="" class="btn btn-primary" style="float:right">완료한 TODO 목록</a>
        </p>
        <ul class="list-group" id="todo-list">
            <!-- Todo items will be added here dynamically -->
        </ul>
    </div>

    <script>
        fetch('http://127.0.0.1:8000/todo/')
            .then(response => response.json())
            .then(data => {
                const todoList = document.getElementById('todo-list');
                data.forEach(todo => {
                    const listItem = document.createElement('li');
                    listItem.className = 'list-group-item';

                    const link = document.createElement('a');
                    link.href = 'http://127.0.0.1:8000/todo/${todo.Id}/';  
                    link.textContent = todo.title;

                    listItem.appendChild(link);

                    if (todo.important) {
                        const badge = document.createElement('span');
                        badge.className = 'badge badge-danger';
                        badge.textContent = '!';
                        listItem.appendChild(badge);
                    }

                    const buttonsDiv = document.createElement('div');
                    buttonsDiv.style.float = 'right';

                    const completeButton = document.createElement('a');
                    completeButton.className = 'btn btn-danger';
                    completeButton.textContent = '완료';
                    completeButton.href = '';  // Update with link for completion action

                    const editButton = document.createElement('a');
                    editButton.className = 'btn btn-outline-primary';
                    editButton.textContent = '수정하기';
                    editButton.href = '';  // Update with link for edit action

                    buttonsDiv.appendChild(completeButton);
                    buttonsDiv.appendChild(editButton);

                    listItem.appendChild(buttonsDiv);

                    todoList.appendChild(listItem);
                });
            })
            .catch(error => console.error('Error:', error));
    </script>
</body>
</html>

    <script>
        // Assuming the todo ID is passed as a query parameter
        const queryParams = new URLSearchParams(window.location.search);
        const todoId = queryParams.get('id');

        if (todoId) {
            fetch(`http://127.0.0.1:8000/todo/${todoId}/`)
                .then(response => response.json())
                .then(data => {
                    const todoCard = document.getElementById('todo-card');
                    document.getElementById('todo-title').textContent = data.title;
                    document.getElementById('todo-description').textContent = data.description;

                    todoCard.style.display = 'block';
                })
                .catch(error => {
                    console.error('Error:', error);
                    // Handle errors, e.g., show an error message
                });
        } else {
            // Handle the case where todo ID is not present
        }
    </script>
</body>
</html>
